```
import numpy as np

def rand(*shape):
	"""
	Returns a matrix of random values with given shape
	:param shape: shape of the matrix
	:return: numpy matrix object of random values with given shape
	"""
	if len(shape) == 1:
		m, n = shape[0], shape[0]
	else:
		m, n = shape
	
	return np.matrix(np.random.rand(m, n))
```
