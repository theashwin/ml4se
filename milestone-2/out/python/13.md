    F11 = E11 * D11

    F11 = (np.abs(F11) > epsilon.small) * F11

    Fi00 = np.where(F00!=0, 1/F00**3, 0)
    Fi11 = np.where(F11!=0, 1/F11**3, 0)
    Fi10 = np.where(F00!=0, -(F01 / F00) * Fi11, 0)
    
    c0 = Fi00 * b0
    c1 = Fi10 * b0 + Fi11 * b1

    x0 = E00 * c0 + E01 * c1
    x1 = E01 * c0 + E11 * c1

    return np.array([x0, x1])
