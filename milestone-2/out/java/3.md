public double scal(GraphicPoint pt1, GraphicPoint pt2, GraphicPoint pt3, GraphicPoint pt4) {
    double xProduct = (pt2.x - pt1.x) * (pt4.x - pt3.x);
    double yProduct = (pt2.y - pt1.y) * (pt4.y - pt3.y);
    double scalar = xProduct + yProduct;

}
